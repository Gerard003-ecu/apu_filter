# =============================================================================
# COMPOSE.YAML — APU Filter Stack
# =============================================================================
# Versión:     8.0.0
# Coherencia:  Alineado con Dockerfile.core v7.0, Dockerfile.agent v7.0
#              y start.sh v3.0.0
#
# Cambios vs 7.0:
#   - Corregido healthcheck del agent (sintaxis CMD-SHELL válida)
#   - Healthcheck del agent verifica proceso propio + Redis
#   - Añadido read_only + tmpfs para hardening de contenedores
#   - Añadido security_opt: no-new-privileges en todos los servicios
#   - Redis ejecuta como usuario no-root (uid 999)
#   - Ajustados límites de memoria Redis (overhead calculado)
#   - start_period de core parametrizado con comentario de cálculo
#   - Añadida red interna para aislamiento de Redis
#   - Validación explícita de volúmenes host requeridos
#   - Healthcheck de core con --fail-with-body para diagnóstico
#   - Añadido init: true para manejo correcto de señales
#   - Logging unificado con anclas YAML (DRY)
#   - Variables de entorno documentadas con comentarios de rango/tipo
#
# Requisitos previos:
#   1. Crear .env en la raíz con:  SECRET_KEY=<valor-seguro-generado>
#   2. Asegurar que existan:       ./data/  ./logs/  ./config/
#   3. NUNCA commitear .env (añadir a .gitignore y .dockerignore)
#
# Ejecución:
#   podman-compose up -d            # Inicio normal
#   podman-compose up -d --build    # Reconstruir imágenes
#   podman-compose ps               # Estado de servicios
#   podman-compose logs -f core     # Logs en tiempo real
# =============================================================================

name: apu

# =============================================================================
# ANCLAS YAML — Configuraciones reutilizables (DRY)
# =============================================================================
# Las anclas evitan duplicación y garantizan consistencia.
# Se referencian con <<: *nombre_del_ancla en cada servicio.

x-logging-default: &logging-default
  driver: json-file
  options:
    max-size: "50m"
    max-file: "5"

x-logging-small: &logging-small
  driver: json-file
  options:
    max-size: "25m"
    max-file: "3"

# Opciones de seguridad comunes a todos los contenedores
x-security-defaults: &security-defaults
  security_opt:
    - no-new-privileges:true
  # init: true asegura que el PID 1 sea un init mínimo (tini)
  # que reenvia señales correctamente y cosecha zombies
  init: true

# =============================================================================
# SERVICIOS
# =============================================================================
services:

  # ─────────────────────────────────────────────────────────────────────────
  # REDIS — Broker de mensajes y caché
  # ─────────────────────────────────────────────────────────────────────────
  # Función: Almacén temporal de datos, cola de mensajes entre core y agent.
  # Acceso: Solo desde la red interna (apu_internal). Puerto host solo para
  #         depuración local (bind a 127.0.0.1).
  # ─────────────────────────────────────────────────────────────────────────
  redis:
    image: docker.io/library/redis:7-alpine

    container_name: apu_redis
    hostname: redis

    # Ejecutar como usuario redis (uid 999 en alpine)
    # Evita correr como root dentro del contenedor
    user: "999:999"

    command: >-
      redis-server
      --appendonly yes
      --maxmemory 256mb
      --maxmemory-policy allkeys-lru
      --tcp-backlog 128
      --save 60 1000
      --save 300 100

    # Solo accesible desde localhost (depuración)
    # Comunicación inter-contenedor usa apu_internal
    ports:
      - "127.0.0.1:6379:6379"

    volumes:
      - redis_data:/data

    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 10s

    restart: unless-stopped

    # Filesystem de solo lectura + tmpfs para directorios que Redis necesita
    read_only: true
    tmpfs:
      - /tmp:size=10M,noexec,nosuid

    <<: *security-defaults

    logging: *logging-small

    # Cálculo de límites de memoria:
    #   maxmemory Redis:      256 MB
    #   Overhead del proceso:  ~50 MB (fork para AOF/RDB, buffers de red)
    #   Margen de seguridad:   ~70 MB
    #   Total contenedor:     384 MB (redondeado)
    #   Reservación mínima:   128 MB (inicio + AOF pequeño)
    deploy:
      resources:
        limits:
          memory: 384M
          cpus: "1.0"
        reservations:
          memory: 128M
          cpus: "0.25"

    networks:
      - apu_internal

    stop_grace_period: 15s

  # ─────────────────────────────────────────────────────────────────────────
  # CORE — API Flask con PyTorch CPU
  # ─────────────────────────────────────────────────────────────────────────
  # Función: Servicio principal de inferencia. Expone API REST en :5002.
  # Dependencias: Redis (para caché y comunicación con agent).
  # Modelo: Carga modelos PyTorch al inicio (operación costosa en tiempo).
  # ─────────────────────────────────────────────────────────────────────────
  core:
    image: apu-core:latest
    pull_policy: never

    build:
      context: .
      dockerfile: infrastructure/Dockerfile.core

    container_name: apu_core
    hostname: core

    ports:
      - "5002:5002"

    env_file:
      - .env

    environment:
      # URL de conexión a Redis (service name resuelto por DNS interno)
      REDIS_URL: "redis://redis:6379/0"
      # Desactivar paralelismo de tokenizers (evita deadlocks con Gunicorn fork)
      TOKENIZERS_PARALLELISM: "false"
      # PYTHONUNBUFFERED y PYTHONPATH: definidos en Dockerfile.core v7.0

    depends_on:
      redis:
        condition: service_healthy

    # ── Healthcheck ──
    # Endpoint /health debe retornar HTTP 200 cuando el servicio está listo.
    #
    # Cálculo de start_period:
    #   Carga de modelo PyTorch:  120-180s (CPU, modelos ~500MB)
    #   Inicialización Flask:      5-10s
    #   Warm-up de workers:       30-60s (Gunicorn prefork)
    #   Margen de seguridad:      50s
    #   Total:                    ~300s (5 minutos)
    #
    # Tiempo hasta detección de fallo total:
    #   start_period + (interval × retries) = 300 + (30 × 5) = 450s = 7.5 min
    healthcheck:
      test:
        - "CMD"
        - "curl"
        - "--silent"
        - "--fail"
        - "--max-time"
        - "8"
        - "http://localhost:5002/health"
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 300s

    volumes:
      # Datos persistentes de la aplicación (modelos, datasets)
      - ./data:/app/data:Z
      # Logs de la aplicación (compartido con host para análisis)
      - ./logs:/app/logs:Z
      # Configuración de solo lectura
      - ./config:/app/config:ro,Z

    # Gunicorn graceful shutdown:
    #   --graceful-timeout define cuánto espera Gunicorn a que los workers
    #   terminen requests en curso. stop_grace_period debe ser >= graceful-timeout.
    stop_grace_period: 45s

    restart: unless-stopped

    # Filesystem restringido: solo los volúmenes montados son escribibles
    read_only: true
    tmpfs:
      - /tmp:size=500M,noexec,nosuid
      - /app/.cache:size=200M,noexec,nosuid

    <<: *security-defaults

    logging: *logging-default

    # Cálculo de límites de memoria:
    #   Modelo PyTorch en memoria:   1.0-2.0 GB
    #   Gunicorn workers (×2):       0.5-1.0 GB
    #   Overhead Python + librerías: 0.3-0.5 GB
    #   Margen de seguridad:         0.5-1.0 GB
    #   Total contenedor:           4 GB
    #   Reservación mínima:         1.5 GB (arranque sin modelo)
    deploy:
      resources:
        limits:
          memory: 4G
          cpus: "4.0"
        reservations:
          memory: 1536M
          cpus: "1.0"

    networks:
      - apu_external
      - apu_internal

  # ─────────────────────────────────────────────────────────────────────────
  # AGENT — Orquestador con PyTorch CPU
  # ─────────────────────────────────────────────────────────────────────────
  # Función: Monitorea fuentes de datos, invoca al core para inferencia
  #          y gestiona el ciclo de vida de las peticiones.
  # Dependencias: Core (API), Redis (estado y colas).
  # Patrón: Loop infinito con CHECK_INTERVAL configurable.
  # ─────────────────────────────────────────────────────────────────────────
  agent:
    image: apu-agent:latest
    pull_policy: never

    build:
      context: .
      dockerfile: infrastructure/Dockerfile.agent

    container_name: apu_agent
    hostname: agent

    env_file:
      - .env

    environment:
      # URL del servicio Core (usa service name, NO container_name)
      CORE_API_URL: "http://core:5002"
      # Conexión a Redis
      REDIS_URL: "redis://redis:6379/0"
      # Timeout para requests HTTP al core (segundos)
      # Rango válido: 5-120. Default: 30.
      REQUEST_TIMEOUT: "30"
      # Intervalo de muestreo del loop principal (segundos)
      # Rango válido: 1-3600. Default: 1 (alta frecuencia / real-time).
      # Valores bajos aumentan carga de CPU. Ajustar según caso de uso.
      CHECK_INTERVAL: "1"
      # Nivel de logging del agent
      # Valores válidos: DEBUG, INFO, WARNING, ERROR, CRITICAL
      LOG_LEVEL: "INFO"
      # PYTHONUNBUFFERED y PYTHONPATH: definidos en Dockerfile.agent v7.0

    depends_on:
      core:
        condition: service_healthy
      redis:
        condition: service_healthy

    # ── Healthcheck ──
    # Estrategia de verificación en dos capas:
    #   1. Verifica que el proceso Python del agent esté vivo (PID file o pgrep)
    #   2. Verifica conectividad con Redis (dependencia crítica del agent)
    #
    # Nota: CMD-SHELL ejecuta el comando a través de /bin/sh -c
    healthcheck:
      test:
        - "CMD-SHELL"
        - >-
          python -c "
          import os, sys;
          import redis;
          r = redis.Redis.from_url('redis://redis:6379/0');
          r.ping();
          sys.exit(0)
          " && pgrep -f 'python' > /dev/null
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 30s

    volumes:
      # Logs del agent (compartido con host)
      - ./logs:/app/logs:Z

    stop_grace_period: 30s

    restart: unless-stopped

    # Filesystem restringido
    read_only: true
    tmpfs:
      - /tmp:size=100M,noexec,nosuid
      - /app/.cache:size=100M,noexec,nosuid

    <<: *security-defaults

    logging: *logging-default

    # Cálculo de límites de memoria:
    #   Runtime Python:            0.3-0.5 GB
    #   Librerías (requests, etc): 0.1-0.3 GB
    #   Datos en tránsito:         0.5-1.0 GB
    #   PyTorch (si aplica):       1.0-2.0 GB
    #   Margen:                    0.5 GB
    #   Total:                     4 GB (conservador)
    #   Reservación:               512 MB
    deploy:
      resources:
        limits:
          memory: 4G
          cpus: "2.0"
        reservations:
          memory: 512M
          cpus: "0.5"

    networks:
      - apu_external
      - apu_internal

# =============================================================================
# REDES
# =============================================================================
# Arquitectura de red:
#
#   ┌──────────────────────────────────────────────────┐
#   │                   HOST                           │
#   │                                                  │
#   │   ┌──────────┐  apu_external  ┌──────────┐      │
#   │   │   CORE   ├────────────────┤  AGENT   │      │
#   │   │  :5002   │                │          │      │
#   │   └────┬─────┘                └────┬─────┘      │
#   │        │       apu_internal        │             │
#   │        │      ┌──────────┐         │             │
#   │        └──────┤  REDIS   ├─────────┘             │
#   │               │  :6379   │                       │
#   │               └──────────┘                       │
#   └──────────────────────────────────────────────────┘
#
# - apu_external: Comunicación core ↔ agent (API HTTP)
# - apu_internal: Comunicación core/agent ↔ redis (datos/caché)
#
# Redis NO está en apu_external: no es accesible directamente
# desde fuera de la red interna.
networks:
  apu_external:
    driver: bridge
    name: apu_external_network

  apu_internal:
    driver: bridge
    name: apu_internal_network
    # Red interna: no permite acceso desde el host excepto por
    # el port binding explícito de Redis (127.0.0.1:6379)
    internal: true

# =============================================================================
# VOLÚMENES
# =============================================================================
volumes:
  redis_data:
    name: apu_redis_data
    # driver: local es el default; explícito para claridad
    driver: local
