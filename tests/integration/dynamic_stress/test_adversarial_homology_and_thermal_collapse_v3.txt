@dataclass(frozen=True, slots=True)
class HomologyInvariants:
    """
    Invariantes homológicos con validación mejorada.
    
    CORRECCIÓN: Validación simplificada de Euler-Poincaré.
    """
    V: int
    E: int
    beta_0: int
    beta_1: int
    euler_characteristic: int
    
    def __post_init__(self) -> None:
        """
        CORRECCIÓN CRÍTICA: Validación simplificada y clara.
        """
        # Validaciones básicas
        if self.V < 0:
            raise ValueError(f"V = {self.V} < 0")
        if self.E < 0:
            raise ValueError(f"E = {self.E} < 0")
        if self.beta_0 < 0:
            raise ValueError(f"β₀ = {self.beta_0} < 0")
        if self.beta_1 < 0:
            raise ValueError(f"β₁ = {self.beta_1} < 0")
        
        # CORRECCIÓN: Una sola validación de Euler-Poincaré
        chi_expected = self.V - self.E
        chi_from_betti = self.beta_0 - self.beta_1
        
        if self.euler_characteristic != chi_expected:
            raise ValueError(
                f"χ inconsistente: euler_characteristic = {self.euler_characteristic} ≠ "
                f"V - E = {chi_expected}"
            )
        
        if chi_expected != chi_from_betti:
            raise ValueError(
                f"EULER-POINCARÉ VIOLADO:\n"
                f"  • V - E = {chi_expected}\n"
                f"  • β₀ - β₁ = {chi_from_betti}"
            )
    
    def __repr__(self) -> str:
        return (
            f"HomologyInvariants(V={self.V}, E={self.E}, "
            f"β₀={self.beta_0}, β₁={self.beta_1}, χ={self.euler_characteristic})"
        )


@dataclass(frozen=True, slots=True)
class SpectralInvariants:
    """Invariantes espectrales (sin cambios significativos)."""
    eigenvalues: RealVector
    lambda_min: float
    lambda_max: float
    zero_multiplicity: int
    fiedler_value: Optional[float]
    spectral_gap: float
    dimension: int
    
    def __post_init__(self) -> None:
        if self.lambda_min < -EPSILON_SPECTRAL:
            raise ValueError(f"λ_min = {self.lambda_min:.2e} < 0")
        
        if self.lambda_max > _CHUNG_SPECTRAL_UPPER_BOUND + EPSILON_SPECTRAL:
            raise ValueError(f"λ_max = {self.lambda_max:.2e} > 2")
        
        if self.lambda_min > self.lambda_max + EPSILON_SPECTRAL:
            raise ValueError(
                f"λ_min = {self.lambda_min:.2e} > λ_max = {self.lambda_max:.2e}"
            )
        
        if self.zero_multiplicity < 0:
            raise ValueError(f"Multiplicidad inválida: {self.zero_multiplicity}")
        
        if self.dimension != len(self.eigenvalues):
            raise ValueError(
                f"Dimensión inconsistente: {self.dimension} ≠ {len(self.eigenvalues)}"
            )
    
    @property
    def is_connected(self) -> bool:
        """Grafo conexo ⟺ β₀ = 1."""
        return self.zero_multiplicity == 1
    
    @property
    def is_fiedler_defined(self) -> bool:
        """Fiedler definido ⟺ grafo conexo y dim ≥ 2."""
        return self.is_connected and self.dimension >= 2
    
    def __repr__(self) -> str:
        return (
            f"SpectralInvariants(λ∈[{self.lambda_min:.6f}, {self.lambda_max:.6f}], "
            f"mult₀={self.zero_multiplicity}, "
            f"a(G)={self.fiedler_value}, "
            f"dim={self.dimension})"
        )


@dataclass(frozen=True, slots=True)
class ThermalCollapseContext:
    """Contexto de colapso térmico (sin cambios significativos)."""
    system_temperature: float
    financial_inertia: float
    laplacian_fiedler: float
    betti_numbers: Dict[str, int]
    isolated_nodes: List[str]
    ekf_innovation_divergence: bool
    max_lyapunov_exponent: float
    
    def __post_init__(self) -> None:
        if self.system_temperature < 0:
            raise ValueError(f"T = {self.system_temperature} < 0")
        if self.financial_inertia < 0:
            raise ValueError(f"ξ = {self.financial_inertia} < 0")
        if self.laplacian_fiedler < 0:
            raise ValueError(f"a(G) = {self.laplacian_fiedler} < 0")
    
    @property
    def should_trigger_veto(self) -> bool:
        """Verifica conjunción de condiciones de veto."""
        return (
            self.system_temperature >= _CRITICAL_THERMAL_STRESS and
            self.financial_inertia <= _COLLAPSED_FINANCIAL_INERTIA and
            self.laplacian_fiedler < EPSILON_FIEDLER and
            self.ekf_innovation_divergence and
            self.max_lyapunov_exponent > 0
        )
    
    def __repr__(self) -> str:
        return (
            f"ThermalCollapseContext(T={self.system_temperature:.2f}, "
            f"ξ={self.financial_inertia:.4f}, "
            f"a(G)={self.laplacian_fiedler:.6f}, "
            f"λ_max={self.max_lyapunov_exponent:.4f}, "
            f"veto={self.should_trigger_veto})"
        )

def compute_euler_betti_numbers_rigorous(
    graph: Union[nx.Graph, nx.DiGraph],
    *,
    verify_structure: bool = True
) -> HomologyInvariants:
    """
    Calcula números de Betti con validación estructural.
    
    Sin cambios funcionales significativos, solo mejora de comentarios.
    """
    if verify_structure:
        num_selfloops = nx.number_of_selfloops(graph)
        if num_selfloops > 0:
            raise ValueError(f"Estructura no simplicial: {num_selfloops} auto-loops")
        
        if graph.is_multigraph():
            raise ValueError("Estructura no simplicial: multigrafo")
    
    if isinstance(graph, nx.DiGraph):
        graph_undirected = graph.to_undirected()
    else:
        graph_undirected = graph.copy()
    
    V: int = graph.number_of_nodes()
    E: int = graph.number_of_edges()
    
    if V == 0:
        return HomologyInvariants(V=0, E=0, beta_0=0, beta_1=0, euler_characteristic=0)
    
    if isinstance(graph, nx.DiGraph):
        beta_0: int = nx.number_weakly_connected_components(graph)
    else:
        beta_0: int = nx.number_connected_components(graph_undirected)
    
    euler_characteristic: int = V - E
    beta_1: int = max(0, E - V + beta_0)
    
    return HomologyInvariants(
        V=V,
        E=E,
        beta_0=beta_0,
        beta_1=beta_1,
        euler_characteristic=euler_characteristic
    )


def compute_normalized_laplacian_spectrum_rigorous(
    graph: Union[nx.Graph, nx.DiGraph],
    *,
    verify_symmetry: bool = True
) -> SpectralInvariants:
    """
    Calcula espectro del Laplaciano normalizado.
    
    CORRECCIÓN CRÍTICA: Cálculo robusto de Fiedler value.
    """
    if graph.number_of_nodes() == 0:
        raise ValueError("Grafo vacío")
    
    if isinstance(graph, nx.DiGraph):
        undirected = graph.to_undirected()
    else:
        undirected = graph.copy()
    
    n: int = undirected.number_of_nodes()
    
    L_norm: NDArray[np.float64] = nx.normalized_laplacian_matrix(undirected).toarray()
    
    if L_norm.shape != (n, n):
        raise ValueError(f"Dimensión inconsistente: L_norm es {L_norm.shape}, esperado ({n}, {n})")
    
    if verify_symmetry:
        symmetry_error: float = np.max(np.abs(L_norm - L_norm.T))
        if symmetry_error > EPSILON_FLOAT64:
            raise ValueError(f"L_norm no simétrica: error = {symmetry_error:.2e}")
    
    eigenvalues: RealVector = np.linalg.eigvalsh(L_norm)
    eigenvalues = np.sort(eigenvalues)
    
    # Multiplicidad de λ=0
    zero_mask = np.abs(eigenvalues) <= _SPECTRAL_ZERO_TOLERANCE
    zero_multiplicity: int = int(np.sum(zero_mask))
    
    # CORRECCIÓN: Fiedler value como primer eigenvalor NO NULO
    fiedler_value: Optional[float] = None
    if zero_multiplicity == 1 and n >= 2:
        # Buscar primer eigenvalor estrictamente positivo
        positive_eigenvalues = eigenvalues[~zero_mask]
        if len(positive_eigenvalues) > 0:
            fiedler_value = float(np.min(positive_eigenvalues))
    
    # Spectral gap (brecha espectral)
    positive_eigenvalues = eigenvalues[eigenvalues > _SPECTRAL_ZERO_TOLERANCE]
    spectral_gap: float = (
        float(np.min(positive_eigenvalues))
        if len(positive_eigenvalues) > 0
        else 0.0
    )
    
    return SpectralInvariants(
        eigenvalues=eigenvalues,
        lambda_min=float(np.min(eigenvalues)),
        lambda_max=float(np.max(eigenvalues)),
        zero_multiplicity=zero_multiplicity,
        fiedler_value=fiedler_value,
        spectral_gap=spectral_gap,
        dimension=n
    )

class TestAdversarialHomologyExactness:
    """Suite de homología adversarial."""
    
    def test_euler_poincare_exactness_rigorous(self) -> None:
        """Verifica Euler-Poincaré con grafo adversarial."""
        graph = nx.DiGraph()
        graph.add_edges_from([
            ("A", "B"), ("B", "C"), ("C", "A"),
            ("C", "D"), ("D", "A"),
        ])
        
        invariants = compute_euler_betti_numbers_rigorous(graph, verify_structure=True)
        
        assert invariants.V == 4, f"V = {invariants.V} ≠ 4"
        assert invariants.E == 5, f"E = {invariants.E} ≠ 5"
        assert invariants.beta_0 == 1, f"β₀ = {invariants.beta_0} ≠ 1"
        assert invariants.beta_1 == 2, f"β₁ = {invariants.beta_1} ≠ 2"
        assert invariants.euler_characteristic == -1, f"χ = {invariants.euler_characteristic} ≠ -1"
        
        # Verificación explícita
        chi_from_ve = invariants.V - invariants.E
        chi_from_betti = invariants.beta_0 - invariants.beta_1
        assert chi_from_ve == chi_from_betti, (
            f"χ(V-E) = {chi_from_ve} ≠ χ(β₀-β₁) = {chi_from_betti}"
        )
        
        print(f"\n{'='*70}")
        print(f"TEST PASADO: Euler-Poincaré Exacto")
        print(f"  • V = {invariants.V}, E = {invariants.E}")
        print(f"  • β₀ = {invariants.beta_0}, β₁ = {invariants.beta_1}")
        print(f"  • χ = {invariants.euler_characteristic}")
        print(f"{'='*70}\n")
    
    def test_homology_method_consistency(self) -> None:
        """Verifica consistencia entre métodos de cálculo."""
        graph = nx.DiGraph()
        graph.add_edges_from([
            ("A", "B"), ("B", "C"), ("C", "A"),
            ("C", "D"), ("D", "A"),
        ])
        
        # Método 1: Euler
        invariants_euler = compute_euler_betti_numbers_rigorous(graph)
        beta1_euler = invariants_euler.beta_1
        
        # Método 2: Incidencia
        graph_undirected = graph.to_undirected() if isinstance(graph, nx.DiGraph) else graph
        B = nx.incidence_matrix(graph_undirected, oriented=True).toarray()
        rank_B = np.linalg.matrix_rank(B, tol=EPSILON_HOMOLOGY)
        beta1_incidence = graph_undirected.number_of_edges() - rank_B
        
        assert beta1_euler == beta1_incidence, (
            f"Métodos inconsistentes: β₁(Euler) = {beta1_euler}, "
            f"β₁(Incidencia) = {beta1_incidence}"
        )
        
        print(f"  ✓ Consistencia verificada: β₁ = {beta1_euler}")
