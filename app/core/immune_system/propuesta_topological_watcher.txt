@dataclass(frozen=True)
class SpectralDecomposition:
    """
    Descomposición espectral verificada con propiedades certificadas.
    
    Invariantes Matemáticos
    ───────────────────────
    • A = Q·Λ·Qᵀ  (reconstrucción exacta dentro de ε_mach)
    • QᵀQ = I     (ortogonalidad certificada)
    • Λ diagonal  (estructura verificada)
    • λᵢ ∈ ℝ      (realidad para matrices simétricas)
    
    Certificado de Validación
    ─────────────────────────
    • reconstruction_error: ‖A − QΛQᵀ‖_F / ‖A‖_F
    • orthogonality_error:  ‖QᵀQ − I‖_F
    • symmetry_error:       ‖A − Aᵀ‖_F / ‖A‖_F
    """
    eigenvalues: NDArray[np.float64]     # λ₁ ≥ λ₂ ≥ ... ≥ λₙ (ordenados)
    eigenvectors: NDArray[np.float64]    # Columnas ortonormales
    reconstruction_error: float
    orthogonality_error: float
    symmetry_error: float
    
    def __post_init__(self) -> None:
        """Verificación post-construcción de invariantes."""
        assert self.eigenvalues.ndim == 1, "Eigenvalores deben ser 1D"
        assert self.eigenvectors.ndim == 2, "Eigenvectores deben ser 2D"
        assert self.eigenvectors.shape[0] == self.eigenvectors.shape[1], \
            "Matriz de eigenvectores debe ser cuadrada"
        assert self.eigenvalues.shape[0] == self.eigenvectors.shape[0], \
            "Dimensión espectral inconsistente"
        
        # Verificar tolerancias numéricas
        assert self.reconstruction_error < 1e-10, \
            f"Error de reconstrucción excesivo: {self.reconstruction_error:.2e}"
        assert self.orthogonality_error < 1e-10, \
            f"Pérdida de ortogonalidad: {self.orthogonality_error:.2e}"
    
    @property
    def condition_number(self) -> float:
        """κ(A) = |λ_max| / |λ_min| con protección contra singularidad."""
        abs_eigs = np.abs(self.eigenvalues)
        nonzero = abs_eigs[abs_eigs > MIN_EIGVAL_TOL]
        if nonzero.size == 0:
            return float('inf')
        return float(nonzero.max() / nonzero.min())
    
    @property
    def spectral_gap(self) -> float:
        """Gap espectral: λ₁ - λ₂ (separación algebraica)."""
        if len(self.eigenvalues) < 2:
            return float('inf')
        sorted_eigs = np.sort(self.eigenvalues)[::-1]
        return float(sorted_eigs[0] - sorted_eigs[1])
    
    @property
    def rank(self) -> int:
        """Rango numérico: #{λᵢ : |λᵢ| > MIN_EIGVAL_TOL}."""
        return int(np.sum(np.abs(self.eigenvalues) > MIN_EIGVAL_TOL))


def compute_certified_eigen_decomposition(
    matrix: NDArray[np.float64],
    hermitian: bool = True,
) -> SpectralDecomposition:
    """
    Descomposición espectral con certificado de validación completo.
    
    Algoritmo
    ─────────
    1. Verificar simetría (si hermitian=True)
    2. Computar eigvalsh/eig según tipo
    3. Ordenar por magnitud decreciente
    4. Verificar reconstrucción A = QΛQᵀ
    5. Verificar ortogonalidad QᵀQ = I
    6. Emitir certificado inmutable
    
    Parameters
    ----------
    matrix : matriz cuadrada de entrada
    hermitian : True para eigvalsh (simétrica/hermítica), False para eig general
    
    Returns
    -------
    SpectralDecomposition con certificado de validación.
    
    Raises
    ------
    ValueError : matriz no cuadrada o no simétrica (cuando hermitian=True)
    NumericalStabilityError : errores de reconstrucción/ortogonalidad excesivos
    """
    mat = np.asarray(matrix, dtype=np.float64)
    
    if mat.ndim != 2 or mat.shape[0] != mat.shape[1]:
        raise ValueError(
            f"Matriz debe ser cuadrada: shape={mat.shape}"
        )
    
    n = mat.shape[0]
    
    # Verificación de simetría
    if hermitian:
        sym_err = float(np.linalg.norm(mat - mat.T, 'fro'))
        rel_sym_err = sym_err / max(float(np.linalg.norm(mat, 'fro')), EPS)
        
        if rel_sym_err > ALGEBRAIC_TOL:
            raise ValueError(
                f"Matriz no simétrica (hermitian=True): "
                f"‖A−Aᵀ‖_F/‖A‖_F = {rel_sym_err:.2e}"
            )
        
        # Imponer simetría exacta
        mat_sym = (mat + mat.T) * 0.5
        eigenvalues, eigenvectors = np.linalg.eigh(mat_sym)
    else:
        eigenvalues, eigenvectors = np.linalg.eig(mat)
        eigenvalues = eigenvalues.real  # Para matrices reales
        eigenvectors = eigenvectors.real
    
    # Ordenar por magnitud decreciente
    idx = np.argsort(np.abs(eigenvalues))[::-1]
    eigenvalues = eigenvalues[idx]
    eigenvectors = eigenvectors[:, idx]
    
    # Certificado de reconstrucción: A ≈ Q·Λ·Qᵀ
    Lambda = np.diag(eigenvalues)
    reconstructed = eigenvectors @ Lambda @ eigenvectors.T
    
    reconstruction_err = float(np.linalg.norm(mat - reconstructed, 'fro'))
    rel_recon_err = reconstruction_err / max(float(np.linalg.norm(mat, 'fro')), EPS)
    
    # Certificado de ortogonalidad: QᵀQ ≈ I
    orthogonality_err = float(
        np.linalg.norm(eigenvectors.T @ eigenvectors - np.eye(n), 'fro')
    )
    
    return SpectralDecomposition(
        eigenvalues=eigenvalues,
        eigenvectors=eigenvectors,
        reconstruction_error=rel_recon_err,
        orthogonality_error=orthogonality_err,
        symmetry_error=rel_sym_err if hermitian else 0.0,
    )

@dataclass(frozen=True)
class RicciCurvatureTensor:
    """
    Tensor de curvatura de Ricci discreto según Forman-Bakry-Émery.
    
    Definición (Forman 2003)
    ────────────────────────
    Para un grafo G=(V,E), la curvatura de Forman-Ricci en arista (x,y) es:
    
        Ric_F(x,y) = w(x,y)·[1/deg(x) + 1/deg(y)] 
                     - Σ_{z~x, z≠y} w(x,z)/√(deg(x)·deg(z))
                     - Σ_{z~y, z≠x} w(y,z)/√(deg(y)·deg(z))
    
    Para vértices: Ric_F(x) = Σ_{y~x} Ric_F(x,y)
    
    Interpretación Geométrica
    ─────────────────────────
    • Ric > 0: curvatura positiva (expansión local, estabilidad)
    • Ric < 0: curvatura negativa (compresión, inestabilidad)
    • Ric = 0: plano (equilibrio dinámico)
    
    Referencias
    ───────────
    Forman, R. (2003). Bochner's method for cell complexes and combinatorial Ricci curvature.
    Ollivier, Y. (2009). Ricci curvature of Markov chains on metric spaces.
    """
    vertex_curvatures: dict[int, float]
    edge_curvatures: dict[tuple[int, int], float]
    average_curvature: float
    curvature_variance: float
    
    @classmethod
    def from_graph(
        cls,
        manifold: DiscreteManifold,
        edge_weights: Optional[dict[tuple[int, int], float]] = None,
    ) -> "RicciCurvatureTensor":
        """
        Calcula curvatura de Forman-Ricci para un grafo discreto.
        
        Parameters
        ----------
        manifold : estructura combinatoria (vértices + aristas)
        edge_weights : pesos w(e) > 0, default = 1.0 (grafo no pesado)
        
        Returns
        -------
        RicciCurvatureTensor inmutable con curvatura certificada.
        """
        if edge_weights is None:
            edge_weights = {e: 1.0 for e in manifold.edges()}
        
        edge_curvatures: dict[tuple[int, int], float] = {}
        
        for (x, y) in manifold.edges():
            w_xy = edge_weights.get((x, y), edge_weights.get((y, x), 1.0))
            deg_x = manifold.vertex_degree(x)
            deg_y = manifold.vertex_degree(y)
            
            # Término diagonal: ponderación inversa de grados
            diagonal_term = w_xy * (1.0 / deg_x + 1.0 / deg_y)
            
            # Término de triangulación: vecinos comunes
            neighbors_x = {z for (a, b) in manifold.edges() if a == x or b == x} - {y}
            neighbors_y = {z for (a, b) in manifold.edges() if a == y or b == y} - {x}
            
            triangulation_x = sum(
                edge_weights.get((x, z), edge_weights.get((z, x), 0.0)) 
                / np.sqrt(deg_x * manifold.vertex_degree(z))
                for z in neighbors_x
            )
            
            triangulation_y = sum(
                edge_weights.get((y, z), edge_weights.get((z, y), 0.0))
                / np.sqrt(deg_y * manifold.vertex_degree(z))
                for z in neighbors_y
            )
            
            # Curvatura de Forman-Ricci
            ric = diagonal_term - triangulation_x - triangulation_y
            edge_curvatures[(x, y)] = float(ric)
        
        # Agregación a nivel de vértices
        vertex_curvatures: dict[int, float] = {}
        for v in manifold.vertices():
            incident_edges = [
                e for e in manifold.edges() if v in e
            ]
            vertex_curvatures[v] = float(sum(
                edge_curvatures.get(e, edge_curvatures.get((e[1], e[0]), 0.0))
                for e in incident_edges
            ))
        
        # Estadísticas globales
        all_curvatures = list(vertex_curvatures.values())
        avg_curvature = float(np.mean(all_curvatures)) if all_curvatures else 0.0
        var_curvature = float(np.var(all_curvatures)) if all_curvatures else 0.0
        
        return cls(
            vertex_curvatures=vertex_curvatures,
            edge_curvatures=edge_curvatures,
            average_curvature=avg_curvature,
            curvature_variance=var_curvature,
        )
    
    def to_matrix(self, vertices: list[int]) -> NDArray[np.float64]:
        """
        Convierte a matriz de curvatura Ric_ij para integración en flujo.
        
        Para matriz diagonal: Ric_ii = curvatura del vértice i.
        Para matriz completa: Ric_ij = curvatura de arista (i,j) si existe.
        """
        n = len(vertices)
        Ric = np.zeros((n, n), dtype=np.float64)
        
        v_to_idx = {v: i for i, v in enumerate(vertices)}
        
        # Diagonal: curvatura de vértices
        for v, curv in self.vertex_curvatures.items():
            if v in v_to_idx:
                idx = v_to_idx[v]
                Ric[idx, idx] = curv
        
        # Fuera de diagonal: curvatura de aristas (simétrica)
        for (i, j), curv in self.edge_curvatures.items():
            if i in v_to_idx and j in v_to_idx:
                idx_i = v_to_idx[i]
                idx_j = v_to_idx[j]
                Ric[idx_i, idx_j] = curv
                Ric[idx_j, idx_i] = curv  # Simetría
        
        return Ric

class ImplicitEulerIntegrator(RicciFlowIntegrator):
    """
    Integrador implícito de Euler (Backward Euler) para flujo de Ricci.
    
    Ecuación Discreta
    ─────────────────
    G_{n+1} = G_n - 2·Δt·Ric(G_{n+1})
    
    Aproximación Linealizada (para evitar punto fijo completo)
    ──────────────────────────────────────────────────────────
    Ric(G_{n+1}) ≈ Ric(G_n)  [primer orden en Δt]
    
    Propiedades
    ───────────
    • Estabilidad L: incondicionalmente estable para Δt > 0
    • Preservación de simetría: (G+Gᵀ)/2 post-paso
    • Regularización SPD: Tikhonov si λ_min < ε
    """
    
    def step(
        self,
        G_current: NDArray[np.float64],
        Ric_current: NDArray[np.float64],
        dt: float,
    ) -> NDArray[np.float64]:
        """Paso implícito con regularización Tikhonov."""
        if dt <= 0.0:
            raise ValueError(f"Paso temporal debe ser > 0: dt={dt}")
        
        # Paso implícito linealizado
        G_next = G_current - 2.0 * dt * Ric_current
        
        # Imponer simetría exacta
        G_next = (G_next + G_next.T) * 0.5
        
        # Regularización SPD (Tikhonov)
        G_next = _regularize_metric(G_next, min_eig=MIN_EIGVAL_TOL)
        
        return G_next

@dataclass(frozen=True)
class BettiNumbers:
    """
    Números de Betti con verificación de consistencia homológica.
    
    Invariantes Topológicos (Hatcher, Cap. 2)
    ──────────────────────────────────────────
    • β₀ ≥ 1  : componentes conexas (π₀)
    • β₁ ≥ 0  : ciclos 1D independientes (H₁)
    • β₂ ≥ 0  : cavidades 2D (H₂) — opcional para complejos 2D
    
    Relación de Euler-Poincaré
    ──────────────────────────
    χ = Σₖ (-1)^k · βₖ
    
    Para grafos (complejos 1-esqueleto): χ = β₀ - β₁
    Para superficies cerradas: χ = 2 - 2g (g = género)
    """
    beta_0: int  # Componentes conexas
    beta_1: int  # Ciclos 1D
    beta_2: int = 0  # Cavidades 2D (opcional)
    
    def __post_init__(self) -> None:
        """Verificación de invariantes topológicos."""
        if self.beta_0 < 1:
            raise TopologicalInvariantError(
                f"β₀={self.beta_0} inválido: debe ser ≥ 1 (espacio no vacío)"
            )
        if self.beta_1 < 0:
            raise TopologicalInvariantError(
                f"β₁={self.beta_1} inválido: debe ser ≥ 0 (ciclos no-negativos)"
            )
        if self.beta_2 < 0:
            raise TopologicalInvariantError(
                f"β₂={self.beta_2} inválido: debe ser ≥ 0 (cavidades no-negativas)"
            )
    
    @property
    def euler_characteristic(self) -> int:
        """χ = β₀ - β₁ + β₂ (fórmula de Euler-Poincaré)."""
        return self.beta_0 - self.beta_1 + self.beta_2
    
    @property
    def is_tree(self) -> bool:
        """Árbol ⟺ β₀=1 ∧ β₁=0 (conexo y acíclico)."""
        return self.beta_0 == 1 and self.beta_1 == 0
    
    @property
    def is_forest(self) -> bool:
        """Bosque ⟺ β₁=0 (acíclico, múltiples componentes permitidas)."""
        return self.beta_1 == 0
    
    def detect_bifurcation(self, previous: "BettiNumbers") -> Optional[str]:
        """
        Detecta bifurcación cualitativa entre estados topológicos.
        
        Tipos de Bifurcación
        ────────────────────
        • Fragmentación: Δβ₀ > 0 (componente se separa)
        • Fusión:        Δβ₀ < 0 (componentes se unen)
        • Ciclización:   Δβ₁ > 0 (nuevo ciclo formado)
        • Colapso:       Δβ₁ < 0 (ciclo eliminado)
        • Cavitación:    Δβ₂ > 0 (nueva cavidad 2D)
        
        Returns
        -------
        Descripción de bifurcación o None si topología estable.
        """
        delta_0 = self.beta_0 - previous.beta_0
        delta_1 = self.beta_1 - previous.beta_1
        delta_2 = self.beta_2 - previous.beta_2
        
        if delta_0 == 0 and delta_1 == 0 and delta_2 == 0:
            return None  # Topología estable
        
        events = []
        if delta_0 > 0:
            events.append(f"Fragmentación (+{delta_0} componentes)")
        elif delta_0 < 0:
            events.append(f"Fusión ({delta_0} componentes)")
        
        if delta_1 > 0:
            events.append(f"Ciclización (+{delta_1} ciclos)")
        elif delta_1 < 0:
            events.append(f"Colapso de ciclo ({delta_1} ciclos)")
        
        if delta_2 != 0:
            events.append(f"Δβ₂={delta_2:+d} (cavidades)")
        
        return " | ".join(events)


def compute_betti_numbers_from_adjacency(
    adjacency: NDArray[np.int_],
) -> BettiNumbers:
    """
    Calcula números de Betti desde matriz de adyacencia (grafo no dirigido).
    
    Algoritmo
    ─────────
    1. β₀: número de componentes conexas (DFS/BFS)
    2. β₁: |E| - |V| + β₀ (fórmula de Euler para grafos)
    
    Parameters
    ----------
    adjacency : matriz de adyacencia simétrica {0,1}
    
    Returns
    -------
    BettiNumbers certificado con invariantes verificados.
    
    Raises
    ------
    ValueError : matriz no cuadrada o no simétrica
    """
    adj = np.asarray(adjacency, dtype=np.int_)
    
    if adj.ndim != 2 or adj.shape[0] != adj.shape[1]:
        raise ValueError(f"Matriz de adyacencia debe ser cuadrada: shape={adj.shape}")
    
    if not np.allclose(adj, adj.T):
        raise ValueError("Matriz de adyacencia debe ser simétrica (grafo no dirigido)")
    
    n_vertices = adj.shape[0]
    
    # β₀: componentes conexas (mediante DFS)
    visited = np.zeros(n_vertices, dtype=bool)
    beta_0 = 0
    
    def dfs(v: int) -> None:
        visited[v] = True
        for u in range(n_vertices):
            if adj[v, u] and not visited[u]:
                dfs(u)
    
    for v in range(n_vertices):
        if not visited[v]:
            dfs(v)
            beta_0 += 1
    
    # Número de aristas: E = (Σᵢⱼ aᵢⱼ) / 2 (grafo no dirigido)
    n_edges = int(np.sum(adj) // 2)
    
    # β₁: fórmula de Euler para grafos (β₁ = |E| - |V| + β₀)
    beta_1 = n_edges - n_vertices + beta_0
    
    return BettiNumbers(beta_0=beta_0, beta_1=beta_1, beta_2=0)

    """
    Morfismo inmunológico con certificación matemática rigurosa.
    
    Mejoras sobre ImmuneWatcherMorphism
    ───────────────────────────────────
    1. Descomposición espectral certificada de G_μν(t)
    2. Integración de flujo de Ricci con estabilidad L-estable
    3. Verificación de bifurcación topológica con historia dual
    4. Backward error analysis en proyecciones ortogonales
    5. Certificados de invariantes categóricos
    """
    
    __slots__ = (
        "_spectral_cache",
        "_ricci_integrator",
        "_betti_history",
        "_bifurcation_log",
    )
    
    def __init__(self, *args, **kwargs) -> None:
        super().__init__(*args, **kwargs)
        
        # Cache de descomposiciones espectrales certificadas
        self._spectral_cache: dict[str, SpectralDecomposition] = {}
        
        # Integrador de flujo de Ricci (implícito por defecto)
        self._ricci_integrator = ImplicitEulerIntegrator()
        
        # Historia de números de Betti (dual de _euler_history)
        self._betti_history: list[BettiNumbers] = []
        
        # Log de bifurcaciones detectadas
        self._bifurcation_log: list[tuple[int, str]] = []
    
    def _evolve_metric_with_certification(
        self,
        telemetry: dict[str, Any],
        dt: float = 0.01,
    ) -> dict[str, SpectralDecomposition]:
        """
        Evoluciona G_μν(t) con certificación espectral completa.
        
        Returns
        -------
        dict de SpectralDecomposition por cada tensor evolucionado.
        """
        # Computar curvatura de Ricci (versión mejorada con topología)
        Ric_P, Ric_T, Ric_Th = self._compute_ricci_from_topology(telemetry)
        
        # Evolución por flujo de Ricci con integrador certificado
        for key, Ric in [
            ("G_PHYSICS", Ric_P),
            ("G_TOPOLOGY", Ric_T),
            ("G_THERMODYNAMICS", Ric_Th),
        ]:
            G_current = self._metric_tensors_state[key]
            
            # Paso de integración
            G_next = self._ricci_integrator.step(G_current, Ric, dt)
            
            # Actualizar estado
            self._metric_tensors_state[key] = G_next
            
            # Certificación espectral
            self._spectral_cache[key] = compute_certified_eigen_decomposition(
                G_next, hermitian=True
            )
        
        return self._spectral_cache
    
    def _compute_ricci_from_topology(
        self,
        telemetry: dict[str, Any],
    ) -> tuple[NDArray, NDArray, NDArray]:
        """
        Calcula tensores de Ricci usando invariantes topológicos reales.
        
        Mejora sobre _compute_discrete_ricci_curvature
        ───────────────────────────────────────────────
        • Usa BettiNumbers certificados en lugar de valores brutos
        • Computa curvatura de Forman-Ricci si se proporciona grafo
        • Penalización cuadrática de ciclos anómalos
        """
        # Extraer números de Betti certificados
        beta_0 = int(telemetry.get('beta_0', 1))
        beta_1 = int(telemetry.get('beta_1', 0))
        
        try:
            betti = BettiNumbers(beta_0=beta_0, beta_1=beta_1)
        except TopologicalInvariantError as exc:
            logger.error(f"Invariantes topológicos inválidos: {exc}")
            # Fallback seguro: topología trivial
            betti = BettiNumbers(beta_0=1, beta_1=0)
        
        # Añadir a historia
        self._betti_history.append(betti)
        
        # Detectar bifurcación
        if len(self._betti_history) >= 2:
            bifurcation = betti.detect_bifurcation(self._betti_history[-2])
            if bifurcation:
                self._bifurcation_log.append((self._evaluation_count, bifurcation))
                logger.warning(
                    f"🔄 Bifurcación topológica #{self._evaluation_count}: {bifurcation}"
                )
        
        # Curvatura topológica (penalización por ciclos y fragmentación)
        Ric_T = np.zeros((2, 2), dtype=np.float64)
        
        # Penalización por fragmentación: Ric_00 ∝ -(β₀ - 1)²
        if betti.beta_0 > 1:
            Ric_T[0, 0] = -1.0 * (betti.beta_0 - 1) ** 2
        
        # Penalización por ciclos: Ric_11 ∝ -β₁²
        if betti.beta_1 > 0:
            Ric_T[1, 1] = -0.5 * (betti.beta_1 ** 2)
        
        # Física y termodinámica: sin cambios (heredados de original)
        Ric_P = np.zeros((3, 3), dtype=np.float64)
        Ric_Th = np.zeros((2, 2), dtype=np.float64)
        
        return Ric_P, Ric_T, Ric_Th
    
    def get_spectral_diagnostics(self) -> dict[str, Any]:
        """
        Diagnóstico espectral de métricas evolucionadas.
        
        Returns
        -------
        dict con:
        • condition_numbers: κ(G) para cada tensor
        • spectral_gaps: gap entre primeros eigenvalores
        • reconstruction_errors: errores de reconstrucción QΛQᵀ
        • ranks: rangos numéricos
        """
        return {
            tensor_name: {
                "condition_number": decomp.condition_number,
                "spectral_gap": decomp.spectral_gap,
                "reconstruction_error": decomp.reconstruction_error,
                "orthogonality_error": decomp.orthogonality_error,
                "rank": decomp.rank,
            }
            for tensor_name, decomp in self._spectral_cache.items()
        }
    
    def get_bifurcation_history(self) -> list[tuple[int, str]]:
        """
        Retorna log completo de bifurcaciones detectadas.
        
        Returns
        -------
        Lista de (evaluation_number, bifurcation_description).
        """
        return list(self._bifurcation_log)
